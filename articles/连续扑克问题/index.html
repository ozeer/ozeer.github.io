连续扑克问题

By ozeer

October 29, 2022

问题：从若干副扑克牌中随机抽 5 张牌，判断是不是一个顺子，即这5张牌是不是连续的。2～10为数字本身，A为1，J为11，Q为12，K为13，而大、小王为 0 ，可以看成任意数字。A 不能视为 14。 示例：输入: [1,2,3,4,5] 输出: True    输入: [0,0,1,2,5] 输出: True

Golang代码：
package main

import (
	"fmt"
	"sort"
)

func isContinuous(a []int) bool {
	pockerNum := len(a)

	sort.Ints(a)
	num := 0
	for i := 0; i < pockerNum; i++ {
		if a[i] == 0 {
			num++
		}
	}

	fmt.Println("王的初始个数：", num)
	// 从非0下标位置开始比较(下标最多比较到4，因为有n+1，所以需要减1)
	for i := num; i < pockerNum-1; i++ {
		// 每次比较目前遍历的扑克和下一张扑克的点数差距
		// 出现两个重复的，一定不是顺子
		if a[i] == a[i+1] {
			return false
		}

		// 如果不相等，比较后一个和前一个的差值，如果差值不等于1，证明需要王来补位置
		if a[i] != a[i+1]-1 {
			// 计算需要(a[i+1] - a[i] + 1)个王来补位置
			num = num - (a[i+1] - a[i]) + 1
		}
	}

	// 最后剩下的大小王刚好为0或者大于0，那么该扑克是顺子，否则如果小于0，证明有两个元素的中间差-num个王来补位置
	fmt.Println("剩余的王个数：", num)
	return num >= 0
}

func main() {
	a := []int{1, 2, 4, 0, 0, 6}
	fmt.Println(isContinuous(a))
}
PHP代码：
<?php

function checkSZ(array $data): bool
{
	sort($data);
	$total = count($data);

	$wNum = 0;
	foreach ($data as $iValue) {
		if ($iValue === 0) {
			$wNum++;
		}
	}

	echo "王的初始数目：$wNum\n";

	for ($i = $wNum; $i < $total - 1; $i++) {
		if ($data[$i] === $data[$i + 1]) {
			return false;
		}

		if ($data[$i+1] - $data[$i] !== 1) {
			$wNum -= ($data[$i + 1] - $data[$i] - 1);
		}
	}

	echo "剩余王的数目：$wNum\n";

	return $wNum >= 0;
}

//$data = [1, 0, 3, 4, 6];
$data = [1, 2, 4, 0, 0, 6];
$res = checkSZ($data);
